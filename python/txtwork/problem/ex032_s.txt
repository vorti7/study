import java.util.Scanner;
public class Main{
	
	public static void main(String[] args) {
		
		Scanner scanner = new Scanner(System.in);
		
		System.out.print("정수를 입력하세요: ");
        int n = Integer.parseInt(scanner.nextLine());
        int sum = 0;
        
        scanner.close();
        
        while (n != 0)
        {
            sum += n % 10;
            n /= 10;
        }
        System.out.print("정수 각 자리수의 합은: " + sum);
	}
}
