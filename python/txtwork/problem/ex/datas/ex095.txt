public class Main{
	public static void main(String[] args) {
		String[] strings = { "C#.", ".NET", "Hello!", "Java.",
                "Python.", "etc" };
		for (String value : strings)
		{
			boolean endsInPeriod = value.endsWith(".");
			System.out.println();
			System.out.printf("'%s': %b\n", value, endsInPeriod);
		}
	}
}
