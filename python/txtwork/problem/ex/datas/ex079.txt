import java.util.Scanner;
public class Main{
	public static void main(String[] args) {
				Scanner scanner = new Scanner(System.in);
				int i, j, rows, k = 1;
        System.out.print("행의 수 : ");
        rows = scanner.nextInt();
        for (i = 1; i <= rows; i++)
        {
            for (j = 1; j <= i; j++)
                System.out.printf("%d ", k++);
            System.out.println();
        }
	}
}
