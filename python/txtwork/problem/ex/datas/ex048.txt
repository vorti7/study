import java.util.Arrays;
public class Main{
	public static void main(String[] args) {
				int[] nums = { 1, 2, 5, 7, 8 };
        System.out.printf("nums: %s\n", Arrays.toString(nums));
        int h_val = nums[0];
        for (int i = 0; i < nums.length; i++)
        {
            if (nums[i] > nums[0])
                h_val = nums[i];
        }
        System.out.println(h_val);
	}
}
