public class Main{

	public static void main(String[] args) {

				System.out.println("1 ~ 20 사이의 홀수를 한 줄에 하나씩 출력하세요.");

        {
            if (n % 2 != 0)
            {
            	System.out.println(n);
            }
        }
	}

}
