//[출력]

import java.util.Arrays;

public class Main{

	public static void main(String[] args) {

				int i;
        int[] testArray = new int[] { 0, 5, 3, 1, 2 };

        Arrays.sort(testArray);

        for (i = 0; i < testArray.length; i++)
        {
        	System.out.println(testArray[i]);
        }

	}

}
