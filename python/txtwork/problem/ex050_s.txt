import java.util.Arrays;
public class Main{
	public static void main(String[] args) {
		int[] nums = { 2, 4, 7, 8, 6 };
        System.out.printf("원본 배열: %s\n", Arrays.toString(nums));
        System.out.println(EvenOdd(nums));
        
	}
	
	public static boolean EvenOdd(int[] nums)
    {
        for (int n : nums)
        {
            if (n % 2 != 0)
                return true;
        }
        return false;
    }
	
}
